
```
#!/bin/sh
# /etc/init.d/myapp_daemon
# 【定制版】APP后台运行 + 脚本监控崩溃 + 自动重启

# ==================== 配置区 (只改这里) ====================
APP_NAME="myapp"
APP_PATH="/usr/bin/myapp"
APP_ARGS=""          # APP参数
PID_FILE="/tmp/${APP_NAME}.pid"
DAEMON_PID_FILE="/tmp/${APP_NAME}_watch.pid"
LOG_FILE="/tmp/${APP_NAME}_run.log"
# 日志大小限制
MAX_LOG_SIZE=$((1024 * 1024))
# ==========================================================

# 日志清理
clean_log() {
    if [ -f "$LOG_FILE" ] && [ $(stat -c "%s" "$LOG_FILE") -gt $MAX_LOG_SIZE ]; then
        tail -n 100 "$LOG_FILE" > "${LOG_FILE}.tmp" && mv "${LOG_FILE}.tmp" "$LOG_FILE"
    fi
}

# 检查APP是否正在运行
is_app_running() {
    pidof -x "$(basename $APP_PATH)" >/dev/null 2>&1
}

# 启动APP（后台运行！）
start_app() {
    clean_log
    echo "[$(date)] 后台启动 APP" >> "$LOG_FILE"
    # 👇 关键：APP 后台运行 & ，完全脱离终端
    $APP_PATH $APP_ARGS &
}

# 监控循环（后台运行，不占串口）
start_watch() {
    if [ -f "$DAEMON_PID_FILE" ]; then
        echo "监控进程已运行"
        return 0
    fi

    echo "启动 APP 后台监控..."
    # 后台循环监控，绝对不占用串口
    (
        while true
        do
            # 检查APP是否存活
            if ! is_app_running; then
                echo "[$(date)] APP 崩溃，自动重启" >> "$LOG_FILE"
                start_app
            fi
            # 每隔1秒检查一次
            sleep 1
        done
    ) &
    echo $! > "$DAEMON_PID_FILE"
}

# 主启动
start() {
    # 先启动APP
    start_app
    # 再启动监控
    start_watch
    echo "APP 后台运行 + 监控启动完成"
}

# 停止：关闭监控+关闭APP
stop() {
    echo "停止监控和 APP..."
    # 杀监控
    if [ -f "$DAEMON_PID_FILE" ]; then
        kill $(cat "$DAEMON_PID_FILE") 2>/dev/null
        rm -f "$DAEMON_PID_FILE"
    fi
    # 杀APP
    pkill -f "$APP_PATH" 2>/dev/null
    rm -f "$PID_FILE"
    echo "已完全停止"
}

restart() { stop; start; }
status() { is_app_running && echo "APP 运行中" || echo "APP 已停止"; }

case "$1" in
    start|stop|restart|status) $1 ;;
    *) echo "用法: $0 {start|stop|restart|status}" ;;
esac
exit 0
```
